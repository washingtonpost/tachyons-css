## Subscription Theme Hover Fill Colors

<p className="pl-sm mb-sm font-xxs">
  You can apply the theme using <code>.subs-theme</code> class.
</p>
<div className="flex flex-row flex-wrap mt-sm mb-md">
  {data.hoverFillColor
    .filter(i => i.selector.includes("theme"))
    .map(i => {
      const className = i.selector.split(":hover")[0].replace(/","/g, "");
      const classSelector = `${className}`;
      return (
        <div
          key={classSelector}
          className="pa-sm flex-1 flex align-center justify-between flex-column b bc-light-gray ma-sm"
          style={{

            height: 200
          }}
        >
          <div className="font-xxs font-bold">
            .{className.split(".").filter(item => item.includes("hover"))}{" "}

          </div>
          <div>
            <code>{i.propertyValue}</code>
          </div>
          <svg
            className={`pa-sm ${className
              .split(".")
              .join(" ")} transition-colors duration-400 ease-in-out b `}
            xmlns="http://www.w3.org/2000/svg"
            viewBox="0 0 64 64"
            width="64"
            height="64"
          >
            <title>fill svg</title>

          </svg>
        </div>
      );
    })}
</div>
