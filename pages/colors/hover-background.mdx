import Head from "next/head";

# Hover Background Colors

<div className="flex flex-column">
  {data.hoverBgColor
    .filter(i => !i.selector.includes("theme"))
    .map(i => {
      const className = i.selector
        .split(":hover")[0]
        .split(".")[1]
        .replace(",", "");

      const classSelector = `.${className}`;
      return (
        <div key={classSelector} className="pa-sm">
          <div className="font--subhead font-bold">
            {classSelector}{" "}
            <span className="italic font-xs gray-dark">{i.comment}</span>
          </div>
          <div>
            <code>{i.propertyValue}</code>
          </div>
          <div
            tabIndex="0"
            className={`pa-md transition-colors duration-400 ease-in-out ${className.replace(":focus", "")} b `}
          ></div>
        </div>
      );
    })}

</div>
