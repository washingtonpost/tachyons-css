# Hover Colors

<div className="flex flex-column">
  {data.hoverColor
    .filter(i => !i.selector.includes("theme"))
    .map(i => {
      const className = i.selector.split(":hover")[0].split(".")[1];
      const classSelector = `.${className}`;
      return (
        <div key={classSelector} className="pa-sm">
          <div className="font--subhead font-bold">
            {classSelector}{" "}
            <span className="italic font-xs gray-dark">{i.comment}</span>
          </div>
          <div>
            <code>{i.propertyValue}</code>
          </div>
          <div
            className={`bold pa-md transition-colors duration-400 ease-in-out ${className} b `}
          >
            Hover Over me
          </div>
        </div>
      );
    })}
</div>
