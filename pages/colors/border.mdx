# Border Colors

<div className="flex flex-column">
  {data.borderColor
    .filter(i => !i.selector.includes("theme"))
    .map(i => {
      return (
        <div key={i.selector} className="pa-sm">
          <div className="font--subhead font-bold">
            {i.selector.replace(":focus", "")}{" "}
            <span className="italic font-xs gray-dark">{i.comment}</span>
          </div>
          <div>
            <code>{i.propertyValue}</code>
          </div>
          <div
            tabIndex={i.selector.includes("focus") ? "0" : "-1"}
            className={`pa-sm bw-thick b-solid ${i.selector
              .replace(":focus", "")
              .split(".")
              .join(" ")
              .replace(":focus", "")} bold`}
          >
            Democracy Dies in Darkness
          </div>
        </div>
      );
    })}
</div>
